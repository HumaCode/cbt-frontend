# 🗄️ Entity Relationship Diagram (ERD) - Aplikasi CBT Universal

Berikut adalah skema relasi database untuk mendukung sistem Computer-Based Test (CBT) skala universal. Skema ini telah mengintegrasikan arsitektur bawaan dari **Spatie Permission** (untuk RBAC) dan **Spatie Media Library** (untuk manajemen lampiran *file*/*polymorphic*).

## 1. Diagram ERD (Mermaid)
*(Jika dokumen ini dibuka di Notion/GitHub/Obsidian yang mendukung Mermaid.js, kode di bawah ini akan di-*render* menjadi diagram visual).*

```mermaid
erDiagram
    %% Spatie Permission (Simplified)
    users ||--o{ model_has_roles : "has"
    roles ||--o{ model_has_roles : "assigned to"

    %% Spatie Media Library (Polymorphic)
    questions ||--o{ media : "morphMany (images/audio)"
    users ||--o{ media : "morphMany (avatars)"

    %% User & Groups
    users ||--o{ group_user : "belongs to"
    groups ||--o{ group_user : "contains"
    
    %% Item Bank
    categories ||--o{ categories : "parent_id"
    categories ||--o{ questions : "has"
    users ||--o{ questions : "created_by"
    questions ||--o{ question_options : "has"

    %% Assessment Builder
    assessments ||--o{ assessment_group : "assigned to"
    groups ||--o{ assessment_group : "assigned"
    assessments ||--o{ assessment_question : "contains"
    questions ||--o{ assessment_question : "included in"

    %% Test Engine (Transactions)
    assessments ||--o{ assessment_sessions : "has"
    users ||--o{ assessment_sessions : "attempts"
    assessment_sessions ||--o{ session_answers : "records"
    questions ||--o{ session_answers : "answered"
    question_options ||--o{ session_answers : "selected"

    users {
        bigint id PK
        string name
        string email
        string password
        timestamp created_at
    }

    groups {
        bigint id PK
        string name
        text description
    }

    categories {
        bigint id PK
        bigint parent_id FK
        string name
    }

    questions {
        bigint id PK
        bigint category_id FK
        enum type "pg, essay, likert"
        enum difficulty "easy, medium, hard"
        longtext content_text
        bigint created_by FK
    }

    question_options {
        bigint id PK
        bigint question_id FK
        text option_text
        boolean is_correct
        decimal weight
    }

    assessments {
        bigint id PK
        string title
        datetime start_date
        datetime end_date
        int duration_minutes
        int max_attempts
        boolean randomize_questions
        boolean randomize_options
        decimal passing_grade
    }

    assessment_sessions {
        bigint id PK
        bigint assessment_id FK
        bigint user_id FK
        datetime start_time
        datetime end_time
        enum status "in_progress, completed, force_submitted"
        decimal total_score
    }

    session_answers {
        bigint id PK
        bigint session_id FK
        bigint question_id FK
        bigint selected_option_id FK "Nullable for Essay"
        text answer_text "Nullable for PG"
        boolean is_correct
        decimal score_earned
    }
